def acc(dat, mod):
    t = 0
    n = 0
    with torch.no_grad():
        for i, l in dat:
            i = i.reshape(-1, 28 * 28)
            out = mod(i)
            _, pred = torch.max(out.data, 1)
            n += l.size(0)
            t += (pred == l).sum()
    return t / n
class TorchModel(torch.nn.Module):
    def __init__(self, im, hidden, out):
        super().__init__()
        self.lin1 = torch.nn.Linear(im, hidden)
        self.nl1 = torch.nn.ReLU()
        self.lin2 = torch.nn.Linear(hidden, out)

    def forward(self, im):
        a = self.lin1(im)
        a = self.nl1(a)
        a = self.lin2(a)
        return a

class selfMod:
    def __init__(self, imsize, hidden, nclass, rate):
        super(selfMod, self).__init__()
        self.imsize = imsize
        self.hidden = hidden
        self.nclass = nclass
        self.rate = rate
        self.w1 = torch.distributions.normal.Normal(0, math.sqrt(2 / imsize)).sample((self.imsize, self.hidden))
        self.w2 = torch.distributions.normal.Normal(0, math.sqrt(6 / (hidden + nclass))).sample((self.hidden, self.nclass))
        self.b1 = torch.zeros((1, self.hidden))
        self.b2 = torch.zeros((1, self.nclass))
        self.lin1 = None
        self.nl1 = None
        self.lin2 = None
        self.nl2 = None

    def __call__(self, x):
        self.lin1 = torch.matmul(x, self.w1) + self.b1
        self.nl1 = self.rl(self.lin1)
        self.lin2 = torch.matmul(self.nl1, self.w2) + self.b2
        self.nl2 = self.sm(self.lin2)
        return self.nl2

    def rl(self, x):
        a = torch.nn.ReLU()
        return a(x)

    def sm(self, x):
        return torch.softmax(x, 1)

    def drl(self, x):
        return x > 0

    def backward(self, ims, out, ls):
        dw2 = torch.matmul(self.nl1.transpose(0, 1), (out - ls) / ls.shape[0])
        db2 = ((out - ls) / ls.shape[0]).sum(0)
        dw1 = torch.matmul(ims.transpose(0, 1), torch.mul(torch.matmul((out - ls) / ls.shape[0], self.w2.transpose(0, 1)), self.drl(self.lin1)))
        db1 = (torch.mul(torch.matmul((out - ls) / ls.shape[0], self.w2.transpose(0, 1)), self.drl(self.lin1))).sum(0)
        self.w1 -= self.rate * dw1
        self.w2 -= self.rate * dw2
        self.b1 -= self.rate * db1
        self.b2 -= self.rate * db2

mod = selfMod(28 * 28, 300, 10, 0.1)
for epoch in range(20):

            nl[j, l[j]] = 1
        mod.backward(i, out, nl)
    end = time.time()
    print(f'Epoch {epoch + 1}: accuracy = {acc(traindata, mod)}, time = {end - start}')
