def acc(data, x):
    true = 0
    n = 0
    with torch.no_grad():
        for ims, ls in data:
            outs = x(ims.reshape(-1, 28 * 28))
            _, pred = torch.max(outs.data, 1)
            n += ls.size(dim = 0)
            true += (pred == ls).sum()
    return true / n
class TorchModel(torch.nn.Module):
    def __init__(self, imin, hidden, imout):
        super().__init__()
        self.linone = torch.nn.Linear(imin, hidden)
        self.relu = torch.nn.ReLU()
        self.lintwo = torch.nn.Linear(hidden, imout)

    def forward(self, im):
        res = self.linone(im)
        res = self.relu(res)
        res = self.lintwo(res)
        return res

class selfMod:
    def __init__(self, imsize, hidden, nclass, bsize, rate):
        self.imsize = imsize
        self.hidden = hidden
        self.nclass = nclass
        self.bsize = bsize
        self.rate = rate
        self.wghts = self.initwghts()

    def initwghts(self):
        w1 = torch.randn((self.hidden, self.imsize))
        w2 = torch.randn((self.nclass, self.hidden))
        b1 = torch.zeros((self.hidden, 1))
        b2 = torch.zeros((self.nclass, 1))
        return {"w1": w1, "w2": w2, "b1": b1, "b2": b2}

    def __call__(self, ims):
        linone = (ims @ self.wghts["w1"].transpose(0, 1)) + self.wghts["b1"].transpose(0, 1)
        relu = torch.mul(linone, linone > 0)
        lintwo = (relu @ self.wghts["w2"].transpose(0, 1)) + self.wghts["b2"].transpose(0, 1)
        tryes = torch.softmax(lintwo, 1)
        self.cache = {"linone": linone, "relu": relu, "lintwo": lintwo, "tryes": tryes}
        self.input = ims
        return lintwo

    def back(self, ls):
        newls = torch.zeros_like(self.cache["tryes"])
        for i in range(ls.shape[0]):
            newls[i, ls[i]] = 1
        diff = (self.cache["tryes"] - newls) / ls.shape[0]
        diff_ = torch.mul(diff @ self.wghts["w2"], self.cache["linone"] > 0)
        self.wghts["w1"] -= self.rate * (diff_.transpose(0, 1) @ self.input)
        self.wghts["w2"] -= self.rate * (diff.transpose(0, 1) @ self.cache["relu"])
        self.wghts["b1"] -= self.rate * (diff_.sum(0).unsqueeze(-1))
        self.wghts["b2"] -= self.rate * (diff.sum(0).unsqueeze(-1))

model = selfMod(28 * 28, 300, 10, 64, 0.1)
