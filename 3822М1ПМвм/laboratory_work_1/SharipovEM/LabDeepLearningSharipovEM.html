def readBinFileMatrix(nameFile):
    pathFiles = os.getcwd()
    with gzip.open(pathFiles + '/' +  nameFile,'rb') as f:
        magic, size = struct.unpack(">II", f.read(8))
        nrows, ncols = struct.unpack(">II", f.read(8))
        data = np.frombuffer(f.read(), dtype=np.dtype(np.uint8).newbyteorder('>'))
        data = data.reshape((size, nrows, ncols))
    return np.array(data)

def ReLU(x):
    return np.maximum(x, 0)

# –æ–ø—Ä–µ–¥–µ–ª–µ–Ω–∏–µ –ø—Ä–æ–∏–∑–≤–æ–¥–Ω–æ–π —Ñ—É–Ω–∫—Ü–∏–∏ –∞–≤—Ç–∏–≤–∞—Ü–∏–∏ ReLU
def ReLuDerivative(x):
    x[x <= 0] = 0
    x[x > 0] = 1
    return x

# –æ–ø—Ä–µ–¥–µ–ª–µ–Ω–∏–µ —Ñ—É–Ω–∫—Ü–∏–∏ –∞–∫—Ç–∏–≤–∞—Ü–∏–∏ softmax –Ω–∞ –≤—ã—Ö–æ–¥–Ω–æ–º —Å–ª–æ–µ
def softmax(x):
    exp = np.exp(x)
    return exp / np.sum(exp, axis = 1, keepdims = True)

# –æ–ø—Ä–µ–¥–µ–ª–µ–Ω–∏–µ —Ñ—É–Ω–∫—Ü–∏–∏ –æ—à–∏–±–∫–∏ –∫—Ä–æ—Å—Å-—ç–Ω—Ç—Ä–æ–ø–∏—è
def crossEntropyLoss(x1, x2):
    return np.mean(-np.sum(x1 * np.log(x2), axis=1))

# —Ñ—É–Ω–∫—Ü–∏—è –ø–æ–¥—Å—á–µ—Ç–∞ —Ç–æ—á–Ω–æ—Å—Ç–∏ –Ω–∞ —Ç–µ—Å—Ç–æ–≤–æ–π –∏–ª–∏ –æ–±—É—á–∞—é—â–µ–π –≤—ã–±–æ—Ä–∫–µ 
def accuracy(x1, x2):
  return np.mean(np.argmax(x1, axis=1) == np.argmax(x2, axis=1))

        # –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –Ω–µ–π—Ä–æ–Ω–æ–≤ –≤—ã—Ö–æ–¥–Ω–æ–≥–æ —Å–ª–æ—è
        self.output_layer = output_layer
        
        # w1, w2 (–≤–µ—Å–∞ —Å–∏–Ω–∞–ø—Ç–∏—á–µ—Å–∫–∏—Ö —Å–≤—è–∑–µ–π) –∏–Ω–∏—Ü–∏–∞–ª–∏–∑–∏—Ä—É–µ–º –Ω–æ—Ä–º–∞–ª—å–Ω—ã–º —Ä–∞—Å–ø—Ä–µ–¥–µ–ª–µ–Ω–∏–µ–º —Å –¥–∏—Å–ø–µ—Ä—Å–∏–µ–π sqrt(input_layer)
        # w1 –∏ b1 –º–∞—Å—Å–∏–≤—ã –¥–ª—è —Ö—Ä–∞–Ω–µ–Ω–∏—è –≤–µ—Å–æ–≤ –∏ —Å–º–µ—â–µ–Ω–∏–π –ø–µ—Ä–≤–æ–≥–æ —Å–ª–æ—è
        self.w1 = np.random.randn(input_layer, hidden_layer) / np.sqrt(input_layer)
        self.b1 = np.zeros((1, hidden_layer))
        # w2 –∏ b2 –º–∞—Å—Å–∏–≤—ã –¥–ª—è —Ö—Ä–∞–Ω–µ–Ω–∏—è –≤–µ—Å–æ–≤ –∏ —Å–º–µ—â–µ–Ω–∏–π –≤—Ç–æ—Ä–æ–≥–æ —Å–ª–æ—è
        self.w2 = np.random.randn(hidden_layer, output_layer) / np.sqrt(hidden_layer)

    # –í–æ–∑–≤—Ä–∞—â–∞–µ—Ç –≤—ã—Ö–æ–¥–Ω–æ–π —Å–∏–≥–Ω–∞–ª –ø–µ—Ä–≤–æ–≥–æ –∏ –≤—Ç–æ—Ä–æ–≥–æ —Å–ª–æ—è
    def forward(self, x):
        self.z1 = np.matmul(x, self.w1) + self.b1
        self.a1 = ReLU(self.z1)
        self.z2 = np.matmul(self.a1, self.w2) + self.b2
        self.a2 = softmax(self.z2)

    def backward(self, xTrain, yTrain, learningRate):

        delta3 = (self.a2 - yTrain) / self.a2.shape[0]
        dW2 = (self.a1.T).dot(delta3)

        self.b2 += -learningRate * db2
