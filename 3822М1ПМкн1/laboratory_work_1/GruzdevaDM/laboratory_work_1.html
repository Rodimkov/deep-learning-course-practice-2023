with open("mnist.pkl", 'wb') as f:

print("Save completed.")

DEFAULT_BATCH_SIZE = 16
DEFAULT_EPOCH_COUNT = 20

Создаем нужные функции и класс для сети¶

In [83]:
def relu(x):
    return np.maximum(x, 0)
    
def softmax(x):
    exp = np.exp(x)
    return exp / np.sum(exp, axis = 1, keepdims = True)
    
def derivative_relu(x):
    return np.where(x > 0.0, 1, 0)

def cross_entropy(x, y):
    return np.mean(- 1 * y * np.log(x))

def accuracy(x, y):
    return np.mean(np.argmax(x, axis = 1) == np.argmax(y, axis = 1))

class NeuralNetwork:
    def __init__(self, input_layer=DEFAULT_INPUT_W * DEFAULT_INPUT_H, output_layer=DEFAULT_OUTPUT_COUNT, hidden_layer=DEFAULT_HIDDEN_COUNT, learning_rate=DEFAULT_LEARNING_RATE):
        self.input_layer = input_layer
        self.output_layer = output_layer
        self.hidden_layer = hidden_layer
        self.learning_rate = learning_rate

        self.weights = [np.random.randn(input_layer, hidden_layer) * 0.05, np.random.randn(hidden_layer, output_layer) * 0.05]
        self.biases = [np.zeros((1, hidden_layer)), np.zeros((1, output_layer))]
    
    def forward(self, x):
        self.input_hidden = np.dot(x, self.weights[0]) + self.biases[0]
        self.after_relu = relu(self.input_hidden)
        self.hidden_output = np.dot(self.after_relu, self.weights[1]) + self.biases[1]
        self.after_softmax = softmax(self.hidden_output)
        return self.after_softmax
    
    def backward(self, x, y):
        delta2 = (self.after_softmax - y) / self.after_softmax.shape[0]
        dw2 = np.dot(self.after_relu.T, delta2)
        drelu = derivative_relu(self.input_hidden)
        delta1 = np.dot(delta2, self.weights[1].T) * drelu
        dw1 = np.dot(x.T, delta1)
        dbo = np.sum(delta2, axis = 0, keepdims=True)
        dbh = np.sum(delta1, axis = 0, keepdims=True)
        self.weights[1] -= self.learning_rate * dw2
        self.weights[0] -= self.learning_rate * dw1
        self.biases[0] -= self.learning_rate * dbh
        self.biases[1] -= self.learning_rate * dbo

In [84]:
import time

mnist = NeuralNetwork()

def train(mnist):
    for epoch in range(DEFAULT_EPOCH_COUNT):
        start_time = time.time()
        for batch in range(0, len(x_train), DEFAULT_BATCH_SIZE):
            img = x_train_normalized[batch : batch + DEFAULT_BATCH_SIZE]
            label = l_train_reshaped[batch : batch + DEFAULT_BATCH_SIZE]
            mnist.forward(img)
            mnist.backward(img, label)
        total_time = time.time() - start_time
        result = mnist.forward(x_train_normalized)
        print("Epoch {} took {:.2f}s with {:.5f} accuracy".format(epoch, total_time, accuracy(result, l_train_reshaped)))
