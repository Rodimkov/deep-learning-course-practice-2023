class Model:
    def __init__(self, image_size, num_classes, learning_rate, num_hidden_neurons):
        self.learning_rate = learning_rate
        self.W1 = torch.distributions.normal.Normal(0, math.sqrt(2 / (image_size))).sample((image_size, num_hidden_neurons))
        self.b1 = torch.zeros((1, num_hidden_neurons))
        self.W2 = torch.distributions.normal.Normal(0, math.sqrt(2 / (num_hidden_neurons + num_classes))).sample((num_hidden_neurons, num_classes))
        self.b2 = torch.zeros((1, num_classes))
        
    def _change_weights(self, dw1, db1, dw2, db2):
        self.W1 -= learning_rate * dw1
        self.W2 -= learning_rate * dw2
        self.b1 -= learning_rate * db1
        self.b2 -= learning_rate * db2
        
    def _relu(self, x, der = False):
        if der:
            return x > 0
        r = torch.nn.ReLU()
        return r(x)
    
    def _softmax(self, x):
        return torch.softmax(x, 1)

    def _forward_pass(self, images: torch.Tensor):
        self.z1 = (images @ self.W1) + self.b1
        self.a1 = self._relu(self.z1)
        self.z2 = (self.a1 @ self.W2) + self.b2
        self.result = self._softmax(self.z2)
        return self.result
    
    
    def _backward_pass(self, images, labels):
        new_labels = torch.zeros_like(self.result)
        for i in range(labels.shape[0]):
            new_labels[i, labels[i]] = 1
        
        dz2 = (self.result - new_labels) / labels.shape[0]
        dW2 = self.a1.transpose(0, 1) @ dz2
        db2 = dz2.sum(0)

        dz1 = torch.mul(dz2 @ self.W2.transpose(0, 1), self._relu(self.z1, der = True))
        dW1 = images.transpose(0, 1) @ dz1
        db1 = dz1.sum(0)
        self._change_weights(dW1, db1, dW2, db2)
        
    def get_accuracy(self, data_loader):
        tp = 0
        n = 0
        with torch.no_grad():
            for images, labels in data_loader:
                images = images.reshape(-1, image_size)
                outputs = self._forward_pass(images)

                _, predicted = torch.max(outputs.data, 1)
                n += labels.size(dim=0)
                tp += (predicted == labels).sum()
        return tp / n
        
    def train(self, train_data_loader, test_data_loader, device, num_epochs):
        print(f'Device type: {device}')
        print("Start train")
        for epoch in range(num_epochs):
            time = perf_counter()
            for images, labels in train_data_loader:
                images = images.view(-1, image_size).to(device)
                labels = labels.to(device)
                self._forward_pass(images)
                self._backward_pass(images, labels)
            time = perf_counter() - time
            print(f'Epoch № {epoch + 1}: Accuracy: {self.get_accuracy(train_data_loader)}, Time: {time}')
        print('Accuracy on test data: {}'.format(self.get_accuracy(test_data_loader)))
